\begin{zadanie}

\begin{mybox}{Podzielność}
  Jeśli $P(x)$ jest wielomianem o współczynnikach całkowitych, a $a,b\in\Z$, to wówczas
  $$a-b|P(a)-P(b).$$
\end{mybox}
